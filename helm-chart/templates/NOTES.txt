Thank you for installing {{ .Chart.Name }}.

Your release is named {{ .Release.Name }}.

{{- if .Values.client.enabled }}
To access the client application:

{{- if contains "NodePort" .Values.client.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Release.Name }}-client)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.client.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ .Release.Name }}-client'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Release.Name }}-client --template "{{ "{{ range (index .status.loadBalancer.ingress 0) }}{{ . }}{{ end }}" }}")
  echo http://$SERVICE_IP:{{ .Values.client.service.port }}
{{- else if contains "ClusterIP" .Values.client.service.type }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ .Release.Name }}-client {{ .Values.client.service.port }}:{{ .Values.client.service.port }}
  echo "Visit http://localhost:{{ .Values.client.service.port }} to access the client application"
{{- end }}
{{- end }}
